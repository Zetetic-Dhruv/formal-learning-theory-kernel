\chapter{Introduction}\label{chap:intro}

This document is the mathematical blueprint for a Lean 4 formalization
of the fundamental theorem of statistical learning. It tracks the core
story of the kernel: the five-way equivalence for PAC learning with a
measurability refinement that weakens the standard Borel hypothesis,
the Littlestone characterization for online learning, Gold's theorem
and the mind-change characterization for Gold-style learning, the
three-paradigm separation, and the Borel-analytic separation theorem
that witnesses the strict weakening of the measurability hypothesis.

The formalization targets Lean 4 \texttt{v4.29.0-rc6} against mathlib4
pinned at commit \texttt{fde0cc5}. The repository sits at
\url{https://github.com/Zetetic-Dhruv/formal-learning-theory-kernel},
the per-module API reference at
\url{https://zetetic-dhruv.github.io/formal-learning-theory-kernel/}.
The kernel contains 354 machine-checked theorems in 21{,}522 lines with
zero sorry. Every statement in this blueprint carrying a
\texttt{\textbackslash lean\{\}} annotation corresponds to a proved
declaration in the kernel.

The audience is learning theorists who do not read Lean 4. Informal
statements are self-contained mathematical prose. The
\texttt{\textbackslash lean\{\}} hyperlinks are there for formalization
readers who want to drill into the precise Lean counterparts of each
statement. Both audiences are served by the same document.

Compression via approximate minimax, the measurable batch learner
monad, and PAC-Bayes infrastructure are deferred to blueprint v2,
together with the Baxter multi-task base case and the extended
criteria family (robust PAC, PAC-Bayes, RKHS statistical layer).
Blueprint v2 will be published as a separate document once v1 is
complete.

\chapter{Primitives}\label{chap:primitives}

\section{Concept and concept class}\label{sec:concept-class}

\begin{definition}[Concept and concept class]\label{def:concept-class}
  \lean{Concept, ConceptClass}
  \leanok
  Fix a domain $X$ and a label set $Y$. A \emph{concept} is a function
  $h : X \to Y$. A \emph{concept class} is a set $C \subseteq Y^X$ of
  concepts.
\end{definition}

Throughout this blueprint we work mostly with the Boolean label set
$Y = \{0, 1\}$, identifying $\{0,1\}^X$ with the power set of $X$. All
definitions extend to arbitrary $Y$ unless stated otherwise, and the
kernel formalizes the general case.

\section{Batch learner}\label{sec:batch-learner}

\begin{definition}[Batch learner]\label{def:batch-learner}
  \lean{BatchLearner}
  \leanok
  A \emph{batch learner} over $(X, Y)$ is a bundled structure carrying a
  hypothesis set $H \subseteq Y^X$, a learning function
  $\mathcal{A}$ that takes a finite training sample
  $S : \mathrm{Fin}\,m \to X \times Y$ and returns a hypothesis in $H$,
  and a proof that the output lies in $H$.
\end{definition}

\begin{definition}[Realizable setting]\label{def:realizable}
  The learning problem for a concept class $C$ is \emph{realizable} if
  the target concept $c$ lies in $C$ and the sample is drawn i.i.d.\
  from a distribution $D$ with labels $c(x_i)$. Unless otherwise noted,
  PAC learnability statements in this blueprint are in the realizable
  setting. The kernel also formalizes the agnostic variant and the
  extended criteria in the extended theorems module.
\end{definition}

\section{Sample, loss, and error}\label{sec:sample-loss}

\begin{definition}[I.i.d.\ sample]\label{def:iid-sample}
  For a distribution $D$ on $X$ and a target concept $c$, an
  \emph{i.i.d.\ sample} of size $m$ is a tuple
  $S = ((x_1, c(x_1)), \dots, (x_m, c(x_m)))$ with each $x_i \sim D$
  drawn independently.
\end{definition}

\begin{definition}[Zero-one loss]\label{def:zero-one-loss}
  For $h, c : X \to \{0,1\}$ and $x \in X$, the \emph{zero-one loss} is
  $\ell(h, c, x) = \mathbf{1}[h(x) \neq c(x)]$.
\end{definition}

\begin{definition}[Empirical and true error]\label{def:errors}
  For a hypothesis $h$, a target $c$, a distribution $D$, and a sample
  $S$ of size $m$:
  \begin{align*}
    \err_S(h) &= \frac{1}{m} \sum_{i=1}^{m} \mathbf{1}[h(x_i) \neq c(x_i)]
      && \text{(empirical error)}, \\
    \err_D(h) &= \mathbb{P}_{x \sim D}\!\left[ h(x) \neq c(x) \right]
      && \text{(true error)}.
  \end{align*}
\end{definition}

\chapter{PAC Characterization}\label{chap:pac}

\section{VC dimension, shattering, and the growth function}\label{sec:vc-dim}

\begin{definition}[Shattering]\label{def:shatters}
  \lean{Shatters}
  \leanok
  A finite set $S \subseteq X$ is \emph{shattered} by a concept class
  $C \subseteq \{0,1\}^X$ if for every function $f : S \to \{0,1\}$
  there exists $h \in C$ with $h \restriction_S = f$. Equivalently, $C$
  realizes every Boolean labeling of $S$.
\end{definition}

\begin{definition}[VC dimension]\label{def:vcdim}
  The \emph{VC dimension} of $C$, written $\VCdim(C)$, is the supremum
  of cardinalities of finite sets shattered by $C$:
  \[
    \VCdim(C) = \sup\{ |S| : S \subseteq X \text{ finite and shattered by } C \}.
  \]
  If the supremum is not finite, we set $\VCdim(C) = \infty$.
\end{definition}

\begin{definition}[Growth function]\label{def:growth}
  The \emph{growth function} $\Pi_C(m) : \N \to \N$ is
  \[
    \Pi_C(m) = \sup_{\substack{S \subseteq X \\ |S| = m}}
      \bigl|\, \{ h \restriction_S : h \in C \}\, \bigr|.
  \]
  It counts the maximum number of distinct Boolean labelings of an
  $m$-element subset that the class can realize.
\end{definition}

\section{Sauer-Shelah lemma}\label{sec:sauer-shelah}

The Sauer-Shelah lemma is the combinatorial heart of the uniform
convergence argument. Without it, the growth function on a class with
finite VC dimension could grow exponentially in the sample size; with
it, the growth is polynomial of degree at most $\VCdim(C)$, and the
union bound over growth-function-many effective hypotheses closes the
PAC proof.

\begin{lemma}[Sauer-Shelah]\label{lem:sauer-shelah}
  \lean{sauer_shelah}
  \leanok
  \uses{def:vcdim, def:growth}
  Let $C$ be a concept class with $\VCdim(C) = d < \infty$. Then for
  every $m \geq 0$,
  \[
    \Pi_C(m) \leq \sum_{i=0}^{d} \binom{m}{i}.
  \]
  In particular, $\Pi_C(m) = O(m^d)$ as $m \to \infty$.
\end{lemma}

\begin{proof}[Proof sketch]
  The classical proof is a downward induction on VC dimension via a
  shift operator on the restriction $C \restriction_S$. For each element
  $x \in S$, partition the restrictions into those distinguished at $x$
  and those that are not; the distinguished subclass has VC dimension
  at most $d - 1$, and a short double-counting argument with Pascal's
  identity closes the induction. The kernel formalizes the quantitative
  variant with explicit binomial-sum bounds as
  \lean{sauer_shelah_quantitative}, and the exponential upper bound
  $\Pi_C(m) \leq (em/d)^d$ used downstream by the symmetrization
  argument as \lean{sauer_shelah_exp_bound}.
\end{proof}

\section{The five-way equivalence}\label{sec:fundamental-theorem}

This section is the centerpiece of the blueprint. The fundamental
theorem bundles five a priori distinct characterizations of PAC
learnability into a single equivalence class, and this kernel adds a
measurability refinement to the statement: the Borel hypothesis that
appears in the standard treatment can be weakened to a
null-measurability hypothesis on the uniform convergence bad event.
The weakening is strict, witnessed by the class constructed in
Chapter~\ref{chap:measurability}.

\begin{definition}[Uniform convergence]\label{def:uniform-convergence}
  \lean{HasUniformConvergence}
  \leanok
  A concept class $C$ has the \emph{uniform convergence property} over
  all distributions on $X$ if for every $\eps, \delta \in (0, 1)$ there
  exists a sample complexity $m_0(\eps, \delta)$, independent of the
  distribution, such that for every distribution $D$ on $X$,
  \[
    \mathbb{P}_{S \sim D^{m_0}}
      \!\left[ \sup_{h \in C} \bigl| \err_S(h) - \err_D(h) \bigr| > \eps \right]
      < \delta.
  \]
\end{definition}

\begin{definition}[PAC learnability]\label{def:pac-learnable}
  \lean{PACLearnable}
  \leanok
  A concept class $C$ is \emph{PAC learnable} if there exist a batch
  learner $\mathcal{A}$ and a sample complexity
  $m_0 : (0, 1)^2 \to \N$, both independent of the target distribution,
  such that for every $\eps, \delta \in (0, 1)$, every distribution $D$,
  and every realizable target $c \in C$,
  \[
    \mathbb{P}_{S \sim D^{m_0(\eps, \delta)}}
      \!\left[ \err_D(\mathcal{A}(S)) > \eps \right] < \delta.
  \]
\end{definition}

\begin{theorem}[Five-way equivalence for PAC learning]\label{thm:fundamental-theorem}
  \lean{fundamental_theorem}
  \leanok
  \uses{def:vcdim, def:uniform-convergence, def:pac-learnable,
        lem:sauer-shelah, def:well-behaved-vc-meas-target}
  Let $X$ be an infinite domain and let $C \subseteq \{0,1\}^X$ be a
  concept class carrying \texttt{WellBehavedVCMeasTarget}
  (Definition~\ref{def:well-behaved-vc-meas-target}). The following are
  equivalent.
  \begin{enumerate}[label=(\roman*)]
    \item $\VCdim(C) < \infty$.
    \item $C$ has the uniform convergence property over all
      distributions on $X$.
    \item Empirical risk minimization over $C$ is a PAC learner.
    \item $C$ is PAC learnable.
    \item Every realizable distribution admits a sample complexity
      bound polynomial in $1/\eps$ and $1/\delta$.
  \end{enumerate}
\end{theorem}

\begin{proof}[Proof sketch]
  Forward direction (i) $\Rightarrow$ (ii). Sauer-Shelah
  (Lemma~\ref{lem:sauer-shelah}) gives polynomial growth of the growth
  function $\Pi_C(2m)$. Symmetrization introduces a ghost sample and
  rewrites the supremum over $C$ as a supremum over the finitely many
  effective labelings of the combined sample (see
  Section~\ref{sec:symmetrization-nullmeas}). Hoeffding's inequality
  bounds the deviation at each fixed effective labeling, and the union
  bound over $\Pi_C(2m)$ such labelings produces the uniform
  convergence rate. The symmetrization step uses the
  \texttt{WellBehavedVCMeasTarget} hypothesis to ensure the ghost-gap
  bad event is null-measurable, which is the exact regularity the
  argument requires.

  Chain (ii) $\Rightarrow$ (iii) $\Rightarrow$ (iv) $\Rightarrow$ (v).
  Uniform convergence implies ERM soundness in the realizable setting:
  the empirical risk minimizer has zero empirical error, and the
  uniform convergence bound transfers zero empirical error to
  vanishing true error. The sample complexity bound from (ii) carries
  over to PAC learnability. The polynomial rate (v) is extracted from
  the explicit $m_0(\eps, \delta)$ produced by the union bound: it is
  polynomial in $1/\eps$ and $\log(1/\delta)$, and therefore polynomial
  in $1/\eps$ and $1/\delta$ as claimed.

  Backward direction (v) or (iv) $\Rightarrow$ (i). Contrapositive: if
  $\VCdim(C) = \infty$, then for every $m$ there is a shattered set
  $S_m$ of size $|S_m| > m$. A no-free-lunch argument on $S_m$
  constructs, for any learner and any sample-complexity bound $m$, a
  realizable target and a distribution on $S_m$ under which the
  learner's expected error exceeds a fixed constant with constant
  probability. Taking $|S_m|$ large enough forces the error above any
  prescribed $\eps$ with probability at least $\delta$, contradicting
  PAC learnability.
\end{proof}

\subsection{The NullMeasurableSet refinement}\label{sec:symmetrization-nullmeas}

The proof of (i) $\Rightarrow$ (ii) above goes through a
symmetrization argument that rewrites the bad event
\[
  \mathrm{Bad}_1(C, m, \eps)
    = \left\{ S \in X^m :
      \sup_{h \in C} \bigl| \err_S(h) - \err_D(h) \bigr| > \eps \right\}
\]
in terms of a ghost sample $S'$ and the two-sided comparison event
\[
  \mathrm{Bad}_2(C, m, \eps)
    = \left\{ (S, S') \in X^m \times X^m :
      \sup_{h \in C} \bigl| \err_{S'}(h) - \err_S(h) \bigr| > \eps \right\}.
\]
The probability bound on $\mathrm{Bad}_2$ requires it to be measurable
with respect to the product measure; the symmetrization step is where
regularity of the class enters the proof.

\begin{definition}[WellBehavedVCMeasTarget]\label{def:well-behaved-vc-meas-target}
  \lean{WellBehavedVCMeasTarget}
  \leanok
  A concept class $C$ over a measurable space $(X, \Sigma)$ satisfies
  \emph{WellBehavedVCMeasTarget} if the two-sided ghost-gap bad event
  $\mathrm{Bad}_2(C, m, \eps)$ is a \emph{null-measurable} subset of
  $(X^m \times X^m,\, \Sigma^{\otimes 2m})$ with respect to every
  product probability measure $D^{\otimes 2m}$ arising from a
  distribution $D$ on $X$. Null-measurability means that
  $\mathrm{Bad}_2$ sits in the completion of the product
  $\sigma$-algebra.
\end{definition}

\begin{definition}[KrappWirth well-behaved]\label{def:krapp-wirth-well-behaved}
  \lean{KrappWirthWellBehaved}
  \leanok
  A concept class $C$ satisfies the \emph{KrappWirth well-behaved}
  hypothesis of Krapp and Wirth (2024) if the two-sided ghost-gap bad
  event $\mathrm{Bad}_2(C, m, \eps)$ is a Borel subset of
  $X^m \times X^m$ in the product Borel structure.
\end{definition}

\begin{proposition}[KrappWirth implies WellBehavedVCMeasTarget]\label{prop:krapp-wirth-implies-wb-vc}
  \lean{KrappWirthWellBehaved.toWellBehavedVC}
  \leanok
  \uses{def:well-behaved-vc-meas-target, def:krapp-wirth-well-behaved}
  Every Borel set is null-measurable with respect to any completion of
  a Borel probability measure. Consequently, any concept class carrying
  \texttt{KrappWirthWellBehaved} also carries
  \texttt{WellBehavedVCMeasTarget}.
\end{proposition}

\begin{proof}
  Borel sets lie in the product $\sigma$-algebra by definition, hence
  in its completion with respect to any measure. The inclusion
  $\mathcal{B} \subseteq \mathcal{B}_\mu^\ast$ (Borel into
  completion) is immediate.
\end{proof}

\begin{remark}[Why the refinement matters]\label{rem:why-refinement}
  The standard treatment in Krapp and Wirth (2024) states the
  fundamental theorem under the Borel hypothesis on the ghost-gap bad
  event, matching Definition~\ref{def:krapp-wirth-well-behaved}. This
  kernel observes that the symmetrization step uses only
  null-measurability of the bad event, not Borelness, and that the
  resulting theorem (Theorem~\ref{thm:fundamental-theorem}) is strictly
  stronger: it applies to concept classes whose ghost-gap bad event is
  null-measurable without being Borel.
  Chapter~\ref{chap:measurability} exhibits a concept class whose bad
  event sits precisely in the gap between Borel and null-measurable,
  certifying that the refinement is not cosmetic.
\end{remark}

\section{Forward pointer to the measurability layer}\label{sec:pac-pointer-forward}

The fundamental theorem (Theorem~\ref{thm:fundamental-theorem}) uses
the \texttt{WellBehavedVCMeasTarget} hypothesis as a black box at this
stage. Chapter~\ref{chap:measurability} opens that black box: it
constructs the Borel-parameterized setting in which the ghost-gap bad
event is always \emph{analytic} (a strictly larger class than Borel),
proves that analytic sets are null-measurable with respect to every
Borel probability measure via Choquet capacitability, and exhibits a
concept class (a singleton class over an analytic non-Borel set) whose
bad event is null-measurable but not Borel. This witnesses the strict
weakening announced in Remark~\ref{rem:why-refinement}.

\chapter{Online Characterization}\label{chap:online}

Online learning sits at the intersection of combinatorics and game theory.
A concept class is online learnable if a learner that sees labeled
examples one at a time, predicting each label before seeing it, can bound
the total number of mistakes by a finite quantity independent of the
adversarial ordering of the stream. Littlestone's 1988 characterization
shows that online learnability is equivalent to finiteness of a purely
combinatorial dimension: the depth of the deepest mistake tree the class
can shatter.

\section{Mistake trees and the Littlestone dimension}\label{sec:ldim}

A mistake tree of depth $n$ records the adversarial game tree that a
learner confronts. Each internal node is labeled with a point $x \in X$;
each leaf at depth $n$ corresponds to a complete history of $n$ labeled
examples. A node's two children correspond to the two possible labels the
adversary might reveal at that point.

\begin{definition}[Mistake tree, shattering]\label{def:ltree-shatters}
  \lean{LTree.isShattered}
  \leanok
  A mistake tree $T : \mathrm{LTree}\,X\,n$ of depth $n$ is
  \emph{shattered} by a concept class $C$ if for every root-to-leaf path
  in $T$, there is a hypothesis $h \in C$ consistent with the labels along
  that path. Equivalently, every label sequence realizable by the tree is
  realizable by some hypothesis in $C$.
\end{definition}

\begin{definition}[Littlestone dimension]\label{def:littlestone-dim}
  \lean{LittlestoneDim}
  \leanok
  The \emph{Littlestone dimension} of $C$, written $\LDim(C)$, is the
  supremum over $n \in \N$ and depth-$n$ mistake trees $T$ such that $T$
  is shattered by $C$. The kernel models $\LDim$ as an element of
  $\mathrm{WithBot}(\mathrm{WithTop}\,\N)$ to accommodate the empty-class
  and infinite-dimension cases uniformly.
\end{definition}

The Littlestone dimension plays the same combinatorial role for online
learning that the VC dimension plays for PAC learning. Crucially, the
inequality between the two goes in only one direction: finite Littlestone
dimension implies finite VC dimension, but not conversely. This asymmetry
drives the paradigm separation in Chapter~\ref{chap:separations}.

\begin{lemma}[Littlestone dominates VC]\label{lem:ldim-ge-vcdim}
  \lean{BranchWiseLittlestoneDim_ge_VCDim}
  \leanok
  \uses{def:vcdim, def:littlestone-dim}
  For every concept class $C$ over a domain $X$ with decidable equality,
  $\VCdim(C) \leq \LDim(C)$. Any set shattered by $C$ as a finite set in
  the VC sense can be arranged as a mistake tree that $C$ shatters in the
  branchwise sense.
\end{lemma}

\section{Standard Optimal Algorithm}\label{sec:soa}

Littlestone's Standard Optimal Algorithm (SOA) is a canonical online
learner defined directly in terms of the Littlestone dimension of the
\emph{version space}, the set of hypotheses in $C$ consistent with the
history so far.

\begin{definition}[Version space]\label{def:version-space}
  \lean{versionSpace}
  \leanok
  For a concept class $C$ and a history $h_{1:t} \in (X \times \{0,1\})^t$,
  the \emph{version space}
  $V(C, h_{1:t}) = \{\, c \in C : c(x_i) = y_i \text{ for all } i \leq t\,\}$
  is the set of hypotheses in $C$ consistent with every labeled example
  seen so far.
\end{definition}

\begin{definition}[Standard Optimal Algorithm]\label{def:soa}
  \lean{SOA_predict_eq}
  \leanok
  At round $t+1$, given the history $h_{1:t}$ and the query point
  $x_{t+1}$, the SOA predicts the label $y \in \{0,1\}$ that maximizes
  $\LDim\bigl(V(C, h_{1:t}) \cap \{\,c : c(x_{t+1}) = y\,\}\bigr)$. In
  case of a tie, the algorithm breaks it by a fixed convention (the
  kernel uses the $\geq$ convention on the true branch).
\end{definition}

\begin{lemma}[SOA halves Littlestone dimension]\label{lem:soa-halves}
  \uses{def:soa, def:version-space, def:littlestone-dim}
  On every mistake made by the SOA, the Littlestone dimension of the
  version space strictly decreases. Consequently, after at most
  $\LDim(C)$ mistakes the version space has Littlestone dimension zero
  (at most one element), and the algorithm makes no further errors.
\end{lemma}

\section{The Littlestone characterization}\label{sec:littlestone-characterization}

\begin{theorem}[Littlestone characterization]\label{thm:littlestone-char}
  \lean{littlestone_characterization}
  \leanok
  \uses{def:littlestone-dim, def:soa, lem:soa-halves,
        def:concept-class}
  A concept class $C$ is online learnable if and only if
  $\LDim(C) < \infty$. The forward direction uses a standard adversary
  argument: infinite $\LDim$ produces arbitrarily deep shattered mistake
  trees, and any learner can be forced to err at every level. The
  backward direction is the SOA: finite $\LDim$ bounds the total number
  of mistakes by $\LDim(C)$ via Lemma~\ref{lem:soa-halves}. For nonempty
  $C$ the optimal mistake bound equals $\LDim(C)$ exactly.
\end{theorem}

The kernel formalizes the two directions as separate declarations
\lean{forward_direction} and \lean{backward_direction}, and the equality
of the optimal mistake bound as \lean{optimal_mistake_bound_eq_ldim}.

\chapter{Gold-Style Learning}\label{chap:gold}

Gold-style learning (identification in the limit from text) is the
oldest learning paradigm considered in this blueprint. A class $C$ is
\emph{EX-learnable} if some learner, presented with an infinite text
enumerating the positive instances of a target concept $c \in C$,
eventually converges on a hypothesis equal to $c$ and never changes its
mind after convergence. Gold's 1967 theorem shows that this criterion is
surprisingly restrictive: no text-based learner can identify any class
that contains all finite subsets of the domain together with at least
one infinite concept.

\section{Texts and EX-identification}\label{sec:gold-setup}

\begin{definition}[EX-learnability]\label{def:ex-learnable}
  \lean{EXLearnable}
  \leanok
  A concept class $C$ over a countable domain $X$ is
  \emph{EX-learnable} if there exists a learner $L$ such that for every
  target concept $c \in C$ and every enumeration $(x_0, x_1, \dots)$ of
  $\{\,x : c(x) = 1\,\}$, the sequence of hypotheses $L(x_{0:t})$ output
  by $L$ converges after finitely many mind changes to a hypothesis
  equal to $c$.
\end{definition}

The definition uses a countable domain because the learner's input is an
enumeration of positive instances, and any enumeration must be
countable. The domain and the concept class can both be infinite; the
restriction is only that they be countable.

\section{Gold's theorem}\label{sec:gold-theorem}

\begin{theorem}[Gold 1967]\label{thm:gold}
  \lean{gold_theorem}
  \leanok
  \uses{def:ex-learnable}
  Let $C$ be a concept class over a countable domain $X$ containing all
  indicator functions of finite subsets and at least one concept $c_\infty$
  with infinite support. Then $C$ is not EX-learnable.
\end{theorem}

\begin{proof}[Proof sketch]
  Fix any candidate learner $L$. We construct an adversarial text for
  $c_\infty$ on which $L$ fails to converge to $c_\infty$. Enumerate the
  positive support of $c_\infty$ as $\{a_0, a_1, a_2, \dots\}$, and define
  a sequence of finite subsets $S_0 \subsetneq S_1 \subsetneq \cdots$ by
  interleaving two processes.

  Process (a) adds the next element of the enumeration to the current
  finite set, ensuring that $\bigcup_n S_n = \{\,a_i : i \in \N\,\}$ is
  exactly the support of $c_\infty$. Process (b) appends a prefix of the
  enumeration long enough to trigger $L$ into predicting the indicator
  of the current $S_n$: since $S_n$ is a finite set in $C$, and $L$ is
  assumed to identify every concept in $C$ in the limit, at some finite
  point $L$ must converge to the indicator of $S_n$ on the text of $S_n$
  alone. Process (b) waits for this convergence, then switches back to
  process (a) to extend $S_n$ by one more element.

  The result is an infinite text whose ultimate support is the support
  of $c_\infty$ but which forces $L$ to oscillate between the indicators
  of $S_0, S_1, S_2, \dots$ infinitely often. Since each $S_n$ differs
  from $c_\infty$ on all later elements of the enumeration, $L$ never
  converges to $c_\infty$. This contradicts the assumption that $L$
  identifies $C$ in the limit.
\end{proof}

The kernel's formal proof in \lean{gold_theorem} implements this
construction explicitly: it builds a countable surjection onto the
positive support of $c_\infty$, interleaves it with the enumeration of a
family of finite subsets, and derives the contradiction from
\texttt{List.ext\_getElem} and elementary arithmetic.

\section{Mind change dimension and ordinal bounds}\label{sec:mind-change}

Gold's theorem is a qualitative non-learnability result. The mind change
dimension refines it into a quantitative stratification: for each ordinal
$\alpha$, one can ask whether a learner exists that uses at most $\alpha$
mind changes on any text, where mind changes are counted as a descending
sequence in a well-founded ordinal. Classes with mind change dimension
below $\omega$ correspond to classes where some finite bound on mind
changes suffices. Classes at $\omega$ or above need genuinely transfinite
bookkeeping.

The use of ordinals rather than natural numbers is not cosmetic. A class
with a uniform finite mind-change bound is characterized by conservative
learning and a finite structural dimension. A class with mind-change
dimension $\omega$ is one where no single finite bound works but every
text is closed off eventually along some well-founded chain. The
structural content of the mind-change theorem is that these distinctions
are proper: the ordinal hierarchy refines the yes-no EX-learnability
partition.

\begin{theorem}[Mind change characterization]\label{thm:mind-change-char}
  \lean{mind_change_characterization}
  \leanok
  \uses{def:ex-learnable, thm:gold}
  A concept class $C$ over a countable domain is EX-learnable with a
  uniform mind-change bound of ordinal $\alpha$ if and only if its
  mind-change dimension is at most $\alpha$. The kernel formalizes the
  finite case fully and provides the ordinal-indexed recursion required
  for the transfinite case.
\end{theorem}

The formal statement uses the kernel's ordinal-typed
\lean{VCDim_embed_ordinal} infrastructure from
\texttt{FLT\_Proofs.Complexity.Ordinal} to handle the cross-paradigm
comparison between VC dimension and mind change dimension as a common
ordinal-valued quantity.

\chapter{Three-Paradigm Separations}\label{chap:separations}

The three paradigms of Chapters \ref{chap:pac} through \ref{chap:gold}
are not equivalent. The kernel formalizes all three logical
relationships between them explicitly, with witness classes that
instantiate the non-implications. Two of the separations are
non-implications witnessed by specific concept classes; the third is an
unconditional implication from online learnability to PAC learnability
that uses only the structural relation $\VCdim \leq \LDim$.

The combined theorem \lean{online_pac_gold_separation} bundles the three
relationships as a single statement of the paradigm landscape.

\section{PAC does not imply Online}\label{sec:pac-not-online}

\begin{theorem}[PAC $\not\Rightarrow$ Online]\label{thm:pac-not-online}
  \lean{pac_not_implies_online}
  \leanok
  \uses{def:pac-learnable, def:littlestone-dim, thm:fundamental-theorem,
        thm:littlestone-char}
  There exists a concept class that is PAC learnable but not online
  learnable.
\end{theorem}

\begin{proof}[Construction of the witness]
  Take $X = \N$ and the \emph{threshold class}
  $T = \{\,h_n : n \in \N\,\}$ where $h_n(k) = \mathbf{1}[k \geq n]$.
  Two facts about this class close the proof.

  First, $\VCdim(T) = 1$. Any set of two points $\{a, b\} \subseteq \N$
  with $a < b$ is not shattered by $T$: the labeling that assigns $1$ to
  $a$ and $0$ to $b$ is not realized by any threshold function (the
  kernel verifies this as \lean{threshold_not_shatter_pair}). A
  singleton is shattered (by $h_0$ and $h_{a+1}$). By
  Theorem~\ref{thm:fundamental-theorem}, $T$ is PAC learnable.

  Second, $\LDim(T) = \infty$. For every $n \in \N$ we construct a
  shattered mistake tree of depth $n$ by binary search: the root is
  labeled with the midpoint of an interval $[0, 2^n]$, and each branch
  recursively labels the midpoint of the half-interval corresponding to
  the answer seen so far. Every root-to-leaf path picks out a threshold,
  and every threshold in the range is realized. The kernel verifies
  this via \lean{thresholdTree_shattered} and concludes
  \lean{ldim_threshold_top}. By Theorem~\ref{thm:littlestone-char}, $T$
  is not online learnable.
\end{proof}

The witness is minimal: $\VCdim(T) = 1$ is the smallest nontrivial VC
dimension, so the class is as PAC-easy as possible while being
online-impossible. The gap between the two dimensions is arbitrary.

\section{Gold does not imply PAC}\label{sec:gold-not-pac}

\begin{theorem}[Gold $\not\Rightarrow$ PAC]\label{thm:gold-not-pac}
  \lean{ex_not_implies_pac}
  \leanok
  \uses{def:ex-learnable, def:vcdim, thm:fundamental-theorem, thm:gold}
  There exists a concept class that is EX-learnable (in the Gold sense,
  from text) but not PAC learnable.
\end{theorem}

\begin{proof}[Construction of the witness]
  Take $X = \N$ and let $F$ be the class of all indicators of finite
  subsets of $\N$. This class is EX-learnable by the trivial enumeration
  learner: after seeing the text corresponding to a finite set $S$, the
  learner eventually observes every element of $S$ and converges to the
  indicator of $S$. No mind change is needed after the last new element.

  On the other hand, $\VCdim(F) = \infty$. For every $n \in \N$, the set
  $\{0, 1, \dots, n - 1\}$ is shattered: every subset of this set is
  itself finite and hence has its indicator in $F$. By
  Theorem~\ref{thm:fundamental-theorem}, $F$ is not PAC learnable.
\end{proof}

The asymmetry exhibited by this witness is exactly the one Gold's
theorem sharpens: finite subsets are trivially identifiable from text
but fail the distributional guarantees that PAC demands. The two
learning criteria use orthogonal forms of regularity: enumeration access
versus statistical concentration.

\section{Online implies PAC}\label{sec:online-implies-pac}

\begin{theorem}[Online $\Rightarrow$ PAC, unconditional]\label{thm:online-implies-pac}
  \lean{online_imp_pac}
  \leanok
  \uses{thm:littlestone-char, lem:ldim-ge-vcdim, thm:fundamental-theorem}
  Every online learnable concept class is PAC learnable.
\end{theorem}

\begin{proof}
  If $C$ is online learnable then $\LDim(C) < \infty$ by
  Theorem~\ref{thm:littlestone-char}. By
  Lemma~\ref{lem:ldim-ge-vcdim}, $\VCdim(C) \leq \LDim(C) < \infty$. By
  the fundamental theorem (Theorem~\ref{thm:fundamental-theorem}), $C$
  is PAC learnable.
\end{proof}

The inclusion is strict by Theorem~\ref{thm:pac-not-online}, and the
kernel records the strict inclusion as
\lean{online_strictly_stronger_pac}.

\section{The three-way picture}\label{sec:three-way}

The consolidated relationship between the three paradigms is:
\begin{align*}
  \text{Online} &\subsetneq \text{PAC}, \\
  \text{PAC} &\not\subseteq \text{Online}, \\
  \text{PAC} &\not\subseteq \text{EX}, \\
  \text{EX} &\not\subseteq \text{PAC}.
\end{align*}
The kernel formalizes all four relations as the single statement
\lean{online_pac_gold_separation}, which bundles
Theorem~\ref{thm:online-implies-pac},
Theorem~\ref{thm:pac-not-online}, and
Theorem~\ref{thm:gold-not-pac} into one proved fact about the paradigm
landscape.

\chapter{The Measurability Layer}\label{chap:measurability}

The fundamental theorem (Theorem~\ref{thm:fundamental-theorem}) was
stated above under the \texttt{WellBehavedVCMeasTarget} hypothesis.
This chapter opens that black box. The structure of the chapter is a
three-step argument. Section~\ref{sec:borel-param} introduces the
Borel-parameterized setting and identifies the precise regularity the
symmetrization step requires. Section~\ref{sec:choquet-bridge} shows
that in the Borel-parameterized setting the ghost-gap bad event is
automatically analytic, and that analytic sets are null-measurable via
the Choquet capacitability theorem. Section~\ref{sec:borel-analytic-sep}
exhibits a concept class whose bad event is analytic but not Borel,
certifying that the measurability refinement of the fundamental theorem
is strict.

\section{Borel parameterization and the symmetrization route}\label{sec:borel-param}

The symmetrization argument at the heart of the uniform convergence
proof rewrites the one-sided bad event as the two-sided ghost-gap event
$\mathrm{Bad}_2(C, m, \eps)$ of Section~\ref{sec:symmetrization-nullmeas}.
The probability bound on $\mathrm{Bad}_2$ hinges on applying Hoeffding's
inequality to the growth-function-many effective labelings of the
combined sample, and then controlling the supremum via a union bound.
For this bound to apply, $\mathrm{Bad}_2$ must be measurable with
respect to the product measure.

The regularity the argument actually uses is null-measurability of
$\mathrm{Bad}_2$ with respect to the completion of the product measure,
not Borel measurability with respect to the product Borel structure.
This is a real difference: there exist sets in the completion that are
not Borel, and we will construct one in
Section~\ref{sec:borel-analytic-sep}. The weaker requirement is all the
proof needs, and using it gives a strictly more general theorem.

\begin{definition}[Borel-parameterized concept class]\label{def:borel-param}
  A concept class $C \subseteq \{0,1\}^X$ over a measurable space
  $(X, \Sigma)$ is \emph{Borel-parameterized} by a standard Borel space
  $(\Theta, \mathcal{B}_\Theta)$ if there exists a jointly measurable
  evaluation map $\mathrm{eval} : \Theta \times X \to \{0,1\}$ such that
  $C = \{\, \mathrm{eval}(\theta, \cdot) : \theta \in \Theta \,\}$.
\end{definition}

Borel parameterization is the standard regularity assumption in
descriptive set theory approaches to learning theory. Almost every
concept class considered in practice admits a Borel parameterization:
finite classes, neural networks with measurable weights, decision trees
of bounded depth, thresholds, and so on. The kernel's
Borel-parameterized setting captures exactly this regularity.

\begin{definition}[Parameterized ghost-gap bad event]\label{def:param-bad-event}
  \lean{paramBadEvent}
  \leanok
  \uses{def:borel-param}
  For a Borel-parameterized class with evaluation map $\mathrm{eval}$,
  sample size $m$, gap threshold $\eps$, and ghost-pair $(S, S')$, the
  \emph{parameterized ghost-gap bad event} is
  \[
    \mathrm{Bad}_{\mathrm{eval}}(m, \eps) = \bigcup_{\theta \in \Theta}
    \Bigl\{\, (S, S') :
      \bigl| \err_{S'}(\mathrm{eval}(\theta, \cdot)) - \err_S(\mathrm{eval}(\theta, \cdot)) \bigr| > \eps
    \,\Bigr\}.
  \]
  This is the projection of a jointly Borel event in
  $\Theta \times X^m \times X^m$ onto the last two factors.
\end{definition}

\section{The analytic measurability bridge}\label{sec:choquet-bridge}

\begin{lemma}[Projections of Borel sets are analytic]\label{lem:borel-proj-analytic}
  The projection of a Borel set in a standard Borel space onto one
  factor of a product is an analytic set. This is the defining property
  of analytic sets: they are exactly the projections of Borel sets,
  equivalently the continuous images of Polish spaces.
\end{lemma}

Applied to Definition~\ref{def:param-bad-event}, the lemma gives that
the parameterized ghost-gap bad event is always analytic in
$X^m \times X^m$. The rest of the measurability argument is to show
that analytic sets are null-measurable for every Borel probability
measure. This is the content of the Choquet capacitability theorem.

\begin{theorem}[Choquet capacitability, Kechris 30.13]\label{thm:choquet-capacity}
  \lean{MeasureTheory.AnalyticSet.compactCap_eq}
  \leanok
  Let $\mu$ be a Borel probability measure on a Polish space. For every
  analytic set $A$, the outer measure of $A$ equals the supremum of
  $\mu(K)$ over compact sets $K \subseteq A$. Equivalently, $A$ is
  $\mu$-capacitable: its outer and inner measures agree, so $A$ is in
  the completion of the Borel $\sigma$-algebra with respect to $\mu$.
\end{theorem}

Choquet capacitability is a pure descriptive-set-theory result. Its
proof in the kernel is 416 lines and is independent of learning theory:
it belongs to \texttt{FLT\_Proofs.PureMath.ChoquetCapacity} and is
Mathlib-contributable as a standalone result. The blueprint treats it
as a black-box citation because its proof is too long to reproduce and
too orthogonal to the learning theory story to justify inclusion.

\begin{corollary}[Analytic sets are null-measurable]\label{cor:analytic-null-measurable}
  \lean{analyticSet_nullMeasurableSet}
  \leanok
  \uses{thm:choquet-capacity}
  Every analytic set in a Polish space is null-measurable with respect
  to every Borel probability measure.
\end{corollary}

Applied to the parameterized ghost-gap bad event of
Definition~\ref{def:param-bad-event},
Corollary~\ref{cor:analytic-null-measurable} gives the regularity the
symmetrization step needs.

\begin{proposition}[Borel-parameterized classes satisfy WellBehavedVCMeasTarget]\label{prop:borel-param-implies-wbvc}
  \lean{analyticSet_nullMeasurableSet_ghostPairs}
  \leanok
  \uses{def:borel-param, def:param-bad-event, cor:analytic-null-measurable,
        def:well-behaved-vc-meas-target}
  Every Borel-parameterized concept class carries
  \texttt{WellBehavedVCMeasTarget}. The bad event
  $\mathrm{Bad}_{\mathrm{eval}}(m, \eps)$ is analytic by
  Lemma~\ref{lem:borel-proj-analytic}, hence null-measurable by
  Corollary~\ref{cor:analytic-null-measurable}, hence in the completion
  of the product $\sigma$-algebra.
\end{proposition}

The fundamental theorem (Theorem~\ref{thm:fundamental-theorem})
therefore applies to every Borel-parameterized concept class, including
the classes for which it was originally stated under the stronger Borel
hypothesis.

\section{The Borel-analytic separation}\label{sec:borel-analytic-sep}

Section~\ref{sec:choquet-bridge} showed that the null-measurability
hypothesis is sufficient for the fundamental theorem. This section
shows that it is strictly weaker than the Borel hypothesis, by
constructing a concept class whose ghost-gap bad event is analytic and
null-measurable but not Borel. This is the witness for the strict
weakening announced in Remark~\ref{rem:why-refinement}.

\begin{definition}[Singleton class over a set]\label{def:singleton-class}
  \lean{singletonClassOn}
  \leanok
  For a set $A \subseteq \R$, define the \emph{singleton class over
  $A$} as the concept class on $\R$ given by
  \[
    C_A = \{\,\mathbf{0}\,\} \cup \{\,\mathbf{1}_{\{a\}} : a \in A\,\}
  \]
  where $\mathbf{0}$ is the everywhere-false concept and
  $\mathbf{1}_{\{a\}}$ is the indicator of the point $a$.
\end{definition}

Every hypothesis in $C_A$ is individually measurable: the zero concept
is measurable because it is constant, and each singleton indicator is
measurable because $\{a\}$ is a Borel set. This is verified as
\lean{singletonClassOn_measurable}. The non-trivial question is whether
the bad event associated to $C_A$ inherits Borel structure.

\begin{definition}[Planar witness event]\label{def:planar-witness}
  \lean{planarWitnessEvent}
  \leanok
  For $A \subseteq \R$, the \emph{planar witness event} is
  \[
    W_A = \{\, (x, y) \in \R^2 : y \in A \text{ and } x \neq y \,\}.
  \]
\end{definition}

\begin{lemma}[Bad event as preimage of planar witness]\label{lem:bad-eq-preimage}
  \lean{singleton_badEvent_eq_preimage_planar}
  \leanok
  \uses{def:singleton-class, def:planar-witness}
  For the singleton class $C_A$, the one-step bad event
  $\mathrm{Bad}(C_A, 1, 1/2)$ equals the preimage of the planar witness
  $W_A$ under the sample-pair projection
  $(\mathrm{Fin}\,1 \to \R) \times (\mathrm{Fin}\,1 \to \R) \to \R^2$.
\end{lemma}

The lemma reduces a question about a bad event on
$\R^1 \times \R^1$ to a question about a planar set in $\R^2$. The
following two theorems characterize the planar set.

\begin{theorem}[Planar witness is analytic]\label{thm:planar-analytic}
  \lean{planarWitnessEvent_analytic}
  \leanok
  \uses{def:planar-witness}
  For every analytic set $A \subseteq \R$, the planar witness $W_A$ is
  an analytic subset of $\R^2$.
\end{theorem}

\begin{theorem}[Planar witness fails to be Borel]\label{thm:planar-not-borel}
  \lean{planarWitnessEvent_not_measurable}
  \leanok
  \uses{def:planar-witness}
  There exists an analytic non-Borel set $A \subseteq \R$ for which the
  planar witness $W_A$ is analytic but not Borel. The existence of such
  $A$ is classical: any complete analytic set in $\R$ suffices (for
  example, a specific projection of the complement of a well-founded
  tree), and the projection $A \mapsto W_A$ preserves the non-Borel
  property.
\end{theorem}

\begin{corollary}[Borel-analytic separation]\label{cor:borel-analytic-separation}
  \lean{singleton_badEvent_not_measurable}
  \leanok
  \uses{lem:bad-eq-preimage, thm:planar-analytic, thm:planar-not-borel,
        prop:borel-param-implies-wbvc, def:krapp-wirth-well-behaved,
        def:well-behaved-vc-meas-target}
  There exists a concept class $C$ over $\R$ such that $C$ carries
  \texttt{WellBehavedVCMeasTarget} (via the analytic measurability
  bridge) but $C$ does not carry \texttt{KrappWirthWellBehaved} (because
  its ghost-gap bad event is analytic and not Borel). Consequently, the
  fundamental theorem (Theorem~\ref{thm:fundamental-theorem}) applies
  to $C$ under the kernel's NullMeasurableSet refinement but not under
  the literature's Borel hypothesis.
\end{corollary}

The corollary closes the loop opened in Chapter~\ref{chap:pac}. The
NullMeasurableSet refinement of the fundamental theorem is not
cosmetic: the singleton class $C_A$ over an analytic non-Borel set is a
concrete witness that the Borel hypothesis is strictly stronger than
what the symmetrization proof actually uses. The kernel formalizes this
as a chain of five theorems in
\texttt{FLT\_Proofs.Theorem.BorelAnalyticSeparation}, each one
corresponding to a step in the argument above.

\chapter{Closing Notes and Forward Pointers}\label{chap:closing}

This document covers the core story of the kernel. Three strands are
deferred to a successor blueprint:

\begin{enumerate}
  \item \textbf{Compression.} The kernel formalizes a compression
    scheme for concept classes via an approximate minimax argument
    using multiplicative weights, giving a constructive route to
    compression schemes where the classical Moran-Yehudayoff argument
    uses the exact minimax theorem. This material belongs to a
    dedicated v2 chapter on compression.
  \item \textbf{The measurable batch learner monad.} The kernel
    introduces \texttt{MeasurableBatchLearner}, a regularity axis
    isolating joint learner measurability, and proves that it forms a
    monad closed under Boolean combination, majority vote, piecewise
    interpolation, and countable selection. The first proof that
    non-neural learners (version spaces) satisfy
    \texttt{MeasurableBatchLearner} uses a rectangle decomposition
    that bypasses Kuratowski-Ryll-Nardzewski measurable selection for
    countable families. This material belongs to a dedicated v2
    chapter on learner regularity.
  \item \textbf{PAC-Bayes, extended criteria, and the Baxter
    multi-task base case.} These are additional characterization
    results the kernel formalizes but which sit outside the core story
    of v1. They will be covered as separate strands in v2 (PAC-Bayes
    as a frequentist-Bayesian bridge, the extended criteria as a
    refinement of the measurability hierarchy, and the Baxter base
    case as a multi-task analogue of the fundamental theorem).
\end{enumerate}

The kernel also contributes a proof engineering layer (the typed proof
operad \texttt{TPG\_FLT} and the measurable inner event metaprogram)
that sits orthogonal to the mathematical content. That layer is
documented in a companion artifact at
\url{https://zetetic-dhruv.github.io/formal-learning-theory-kernel/methodology/}
(in preparation) which reuses the same leanblueprint infrastructure and
is cross-linked with this blueprint on the case study of the measurable
inner event pattern used by the symmetrization step of
Chapter~\ref{chap:measurability}.
