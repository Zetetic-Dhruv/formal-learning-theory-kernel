% Shared preamble for print and web versions of the blueprint.

% Fallback definitions for the blueprint package commands.
% In the web build, these are overridden by the `blueprint` package.
% In the print build, they are no-ops so \lean, \uses, \leanok render nothing.
\providecommand{\lean}[1]{}
\providecommand{\leanok}{}
\providecommand{\uses}[1]{}
\providecommand{\proves}[1]{}
\providecommand{\notready}{}
\providecommand{\discussion}[1]{}
\providecommand{\home}[1]{}
\providecommand{\github}[1]{}
\providecommand{\dochome}[1]{}

\newtheorem{proposition}[theorem]{Proposition}
\newtheorem{corollary}[theorem]{Corollary}

\theoremstyle{definition}
\newtheorem{definition}[theorem]{Definition}
\newtheorem{example}[theorem]{Example}
\newtheorem{remark}[theorem]{Remark}
